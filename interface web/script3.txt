document.addEventListener("DOMContentLoaded", function() {
    // Get all the sidebar links
    const links = document.querySelectorAll(".nav_link");
    
    // Attach a click event listener to each link
    links.forEach(function(link) {
      link.addEventListener("click", function(event) {
        event.preventDefault();
        
        // Get the page identifier from the data-page attribute
        const page = link.getAttribute("data-page");
        
        // Generate the file path based on the page identifier
        const filePath =   page + ".php";
        
        // Load the content for the selected page
        loadPageContent(filePath);
      });
    });
    
    // Function to load the content for a specific page
    function loadPageContent(filePath) {
        // Perform an AJAX request or fetch to retrieve the content of the page
        fetch(filePath)
          .then(response => response.text())
          .then(content => {
            // Update the content container with the loaded content
            document.getElementById("content").innerHTML = content;
          })
          .catch(error => {
            console.error("Error loading page:", error);
          });
      }
  });
